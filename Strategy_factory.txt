# athena/experiment/dataset.py
class MonthlyDatasetManager:
    def __init__(self, base_path="data/raw/15m"):
        self.base_path = Path(base_path)
    
    def list_available_months(self, symbol="BTCUSDT") -> List[str]:
        """Возвращает список доступных месяцев по файлам"""
        pattern = f"{symbol}_15m_*.csv"
        files = self.base_path.glob(pattern)
        months = [f.stem.split("_")[-1] for f in files]
        return sorted(months)
    
    def load_month(self, month: str, symbol="BTCUSDT") -> pd.DataFrame:
        """Загружает один месяц"""
        path = self.base_path / f"{symbol}_15m_{month}.csv"
        return pd.read_csv(path)
    
    def load_period(self, months: List[str], symbol="BTCUSDT") -> pd.DataFrame:
        """Загружает несколько месяцев подряд"""
        dfs = [self.load_month(m, symbol) for m in months]
        return pd.concat(dfs, ignore_index=True)
    
    def train_val_test_split(self, train_months, val_months, test_months):
        """Честное временное разделение"""
        return (
            self.load_period(train_months),
            self.load_period(val_months),
            self.load_period(test_months),
        )

# athena/experiment/ablation.py
class AblationMatrix:
    """
    Тестирует комбинации групп фич.
    Группы фич из engineer.py:
    - price: ценовые фичи
    - indicators: RSI, MACD, BB и т.д.
    - orderbook: OBI, spread
    - orderflow: trade_imbalance, execution_ratio
    - multi_horizon: ret_5m, vol_5m и т.д.
    - regime: vol_regime, atr_x_vol
    - rolling_stats: rolling_mean, sharpe, skew
    - volume_dynamics: vol_ratio, vol_zscore
    - time: hour_sin, dow_sin, session
    """
    
    GROUPS = {
        "price": ["ret_1", "ret_2", "ret_3", "ret_5", "ret_10", "ret_20", 
                  "price_pos_20", "price_pos_60"],
        "indicators": ["rsi", "rsi_slope", "bb_pos", "bb_width", "macd_hist", 
                       "stoch_k", "ema_9_dist", "ema_21_dist"],
        "orderbook": ["ob_imb_5", "ob_imb_20", "spread", "ba_ratio", "ob_pressure"],
        "orderflow": ["trade_imbalance", "execution_ratio", "total_liquidity"],
        "multi_horizon": ["ret_5m", "vol_5m", "ret_15m", "vol_15m", "ret_30m", "vol_30m"],
        "regime": ["vol_regime", "atr_x_vol", "vol_x_imbalance"],
        "rolling_stats": ["rolling_sharpe_10", "rolling_sharpe_30", "rolling_skew_10"],
        "volume_dynamics": ["vol_ratio_5", "vol_zscore_5", "vol_momentum_5"],
        "time": ["hour_sin", "dow_sin", "london_open", "ny_open", "weekend"],
    }
    
    def __init__(self, config, engineer, trainer):
        self.config = config
        self.engineer = engineer
        self.trainer = trainer
        self.results = {}
    
    def run(self, train_df, val_df, test_df, base_name="ablation"):
        """Запускает все комбинации выключенных групп"""
        scenarios = self._generate_scenarios()
        
        for name, disabled_groups in scenarios.items():
            print(f"\n🏃 Testing: {name}")
            
            # Отключаем указанные группы фич
            self.engineer.disabled_feature_groups = disabled_groups
            
            # Обучаем модель на train
            model = self.trainer.train_on_df(train_df, val_df)
            
            # Тестируем на test
            metrics = self.trainer.evaluate_on_df(model, test_df)
            
            self.results[name] = {
                "disabled_groups": disabled_groups,
                "metrics": metrics,
                "model_path": f"models/{base_name}_{name}.pkl"
            }
            
            # Сохраняем модель
            self.trainer.save_model(model, self.results[name]["model_path"])
        
        return self.results
    
    def _generate_scenarios(self):
        """Генерирует сценарии: baseline, each_off, all_off"""
        scenarios = {
            "baseline": [],  # ничего не отключено
            "all_on": [],    # дубль baseline для ясности
        }
        
        # По одной группе отключаем
        for group in self.GROUPS.keys():
            scenarios[f"no_{group}"] = [group]
        
        # Ключевые комбинации (как ты и хотел)
        scenarios["no_rolling"] = ["rolling_stats"]
        scenarios["no_sentiment"] = ["sentiment"]  # если есть
        scenarios["no_rolling_sentiment"] = ["rolling_stats", "sentiment"]
        scenarios["no_regime"] = ["regime"]
        scenarios["minimal"] = ["price", "indicators", "orderbook"]  # только базовое
        
        return scenarios

# athena/experiment/registry.py
class ExperimentRegistry:
    def __init__(self, storage_path="data/experiments"):
        self.storage = Path(storage_path)
        self.models_dir = self.storage / "models"
        self.results_dir = self.storage / "results"
        self.models_dir.mkdir(parents=True, exist_ok=True)
        self.results_dir.mkdir(parents=True, exist_ok=True)
    
    def save_experiment(self, name: str, results: Dict, model=None):
        """Сохраняет полный результат эксперимента"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        exp_id = f"{timestamp}_{name}"
        
        # Сохраняем метрики
        results_path = self.results_dir / f"{exp_id}.json"
        with open(results_path, "w") as f:
            json.dump(results, f, indent=2)
        
        # Сохраняем модель (если есть)
        if model:
            model_path = self.models_dir / f"{exp_id}.pkl"
            with open(model_path, "wb") as f:
                pickle.dump(model, f)
        
        # Обновляем метаданные
        self._update_metadata(exp_id, name, results_path, model_path if model else None)
        
        return exp_id
    
    def load_model(self, exp_id: str):
        """Загружает модель по ID эксперимента"""
        model_path = self.models_dir / f"{exp_id}.pkl"
        with open(model_path, "rb") as f:
            return pickle.load(f)
    
    def compare_experiments(self, exp_ids: List[str]) -> pd.DataFrame:
        """Сравнивает несколько экспериментов по Sharpe, PF, Return"""
        rows = []
        for exp_id in exp_ids:
            results_path = self.results_dir / f"{exp_id}.json"
            with open(results_path) as f:
                data = json.load(f)
            rows.append({
                "exp_id": exp_id,
                "sharpe": data.get("sharpe_ratio", 0),
                "profit_factor": data.get("profit_factor", 0),
                "return_pct": data.get("total_return_pct", 0),
                "win_rate": data.get("win_rate", 0),
                "max_dd": data.get("max_drawdown_pct", 0),
            })
        return pd.DataFrame(rows)
